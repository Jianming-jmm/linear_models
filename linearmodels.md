``` r
data('nyc_airbnb')

nyc_airbnb = 
  nyc_airbnb |>

  filter(borough != "Staten Island") |>
```

``` r
nest_lm_res =
  nyc_airbnb |> 
  nest(data = -borough) |> 
  mutate(
    models = map(data, \(df) lm(price ~ stars + room_type, data = df)),
    results = map(models, broom::tidy)) |> 
  select(-data, -models) |> 
  unnest(results)
```

``` r
nest_lm_res |>
```

``` r
manhattan_neighborhood_fit_df = 
nyc_airbnb|>
  filter(
    borough == 'Manhattan'
  )|>
  nest(data = -(borough:neighborhood))|>
  mutate(
    models = map(data, \(df) lm(price ~ stars + room_type, data = df)),
    results = map(models, broom::tidy))|>
  select(-data, -models) |> 
  unnest(results)
```
